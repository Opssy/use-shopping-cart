---
title: removeCartItem()
---

Completely removes a product and quantity that comes with it from the removeCartItem

```jsx
import { useShoppingCart } from 'use-shopping-cart'
import { formatCurrencyString } from '../util'

function Product(product) {
  const { removeCartItem } = useShoppingCart()
  const { name, sku, price, image, currency } = product
  return (
    <div
      style={{
        display: 'flex',
        flexDirection: 'column',
        justifyContent: 'center',
        alignItems: 'center',
      }}
    >
      <img alt={'Alt Tet'} src={image} />
      <div>
        <p>{name}</p>
        <p>{formatCurrencyString({ price: price, currency })}</p>
      </div>
      <button onClick={() => removeCartItem(sku)} backgroundColor={'black'}>
        Remove from Cart
      </button>
    </div>
  )
}
```
