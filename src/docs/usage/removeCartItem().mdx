---
title: removeCartItem()
---

Completely removes a product and quantity that comes with it from the removeCartItem

```jsx
/**@jsx jsx */
import { jsx, Box, Image, Button, Flex } from 'theme-ui';
import { useShoppingCart } from 'use-shopping-cart';
import { toCurrency } from '../util';

const Product = (product) => {
  const { removeCartItem } = useShoppingCart();
  const { name, sku, price, image, currency } = product;
  return (
    <Flex
      sx={{
        flexDirection: 'column',
        justifyContent: 'center',
        alignItems: 'center',
      }}
    >
      <Image src={image} />
      <Box>
        <p>{name}</p>
        <p>{toCurrency({ price: price, currency })}</p>
      </Box>
      <Button onClick={() => removeCartItem(product)} backgroundColor={'black'}>
        Remove from Cart
      </Button>
    </Flex>
  );
};
```
